CREATE TABLE IF NOT EXISTS sensors (sensor_id serial PRIMARY KEY, name VARCHAR (50) NOT NULL, type VARCHAR (50) NOT NULL , UNIQUE (name, type))
CREATE TABLE IF NOT EXISTS locations (location_id serial PRIMARY KEY, location VARCHAR (50))
CREATE TABLE IF NOT EXISTS sensors_locations (s_l_id serial PRIMARY KEY, sensor_id INT NOT NULL, location_id INT NOT NULL, FOREIGN KEY (sensor_id) REFERENCES sensors (sensor_id), FOREIGN KEY (location_id) REFERENCES locations (location_id), UNIQUE (sensor_id, location_id))
CREATE TABLE IF NOT EXISTS logging (log_id serial PRIMARY KEY, s_l_id INT NOT NULL, timestamp timestamp NOT NULL DEFAULT NOW(), value VARCHAR(50) NOT NULL, FOREIGN KEY (s_l_id) REFERENCES sensors_locations (s_l_id))
